#以下分别按照1，2，....标识的段落，依次将R语句拷贝到RGui运行，看测试结果
#--------------------------------------------------1------------------------------------------------
  library(mclust)

#对USArrests这一四维数据集做基于模型聚类，因此结果图会是每一维两两聚类的结果，也可以降维后再做基于模型聚类
  data=read.table("USArrests.txt")
#Mclust提供了两种方法用于评估不同模型在不同分组下的可能性，分别为：
#BIC( Bayesian Information Criterion ): 贝叶斯信息判别标准
#ICL( integrated complete-data likelihood ): 综合完全数据可能性
#--------------------------------------------------2------------------------------------------------
  par(mfrow=c(1,2))	#同时画两张图，c(1,2)就是一行两列，和matlab的subplot类似
  mb=Mclust(data)
#--------------------------------------------------3------------------------------------------------
  print(mb)
#直接输入mb会得到如下信息：
#这里需要对结果稍作解释，第一行告诉我们'Mclust'以VVE模型将数据分为3类。第3行开始，它告诉我们'Mclust'的输出结果中包含了如下内容，我们可以通过$来提取。举个例子，我们提取Mclust的聚类结果和已知结果进行比较
#--------------------------------------------------4------------------------------------------------
#输入1-4可得到不同的结果，输入0退出。
  plot(mb)
#--------------------------------------------------5------------------------------------------------
#输出每个样本点的聚类结果
  mb$class
#输出最优选择模型，即第一张图中对应的使得BIC达到最大的指标
  mb$modelName
#可以查看mclust提供的所有模型，一共提供了14种模型，Mclust默认得到14种模型1到9组的分析结果，然后根据一定的标准选择最终的模型和分组数[1]。
  ?mclustModelNames
#输出最优聚类簇数
  mb$G
#观测值在给定群集中的概率，输出前6个
  head(mb$z)
#得到概率，平均值，方差
  summary(mb, parameters = TRUE)
